\newcommand{\setcounters}[1]
{
  \setcounter{equation}{#1}
  \setcounter{figure}{#1}
  \setcounter{table}{#1}
}

\newcommand{\unit}[1]
{
  \hspace{1mm}\mathrm{[#1]}
}

\newcommand{\degc}
{
  \hspace{1mm}\mathrm{[}{}^\circ\mathrm{C]}
}

\newcommand{\refig}[1]
{
  図\ref{fig::#1}
}

\newcommand{\refeq}[1]
{
  式（\ref{eq::#1}）
}

\newcommand{\reftab}[1]
{
  表\ref{tab::#1}
}

\newcommand{\fig}[5]
{
  \begin{figure}[#1]
    \begin{center}
      \includegraphics[width=#2\hsize]{#3}
    \end{center}
    \caption{#4}
    \label{fig::#5}
  \end{figure}
}

\newcommand{\equ}[2]
{
  \begin{equation}
    #2 \label{eq::#1}
  \end{equation}
}

\newcommand{\eqnv}[4]
{
  \begin{eqnarray}
    #1 \label{eq::#2} \\
    #3 \label{eq::#4}
  \end{eqnarray}
}

\newcommand{\diff}[2]
{
  \frac{\mathrm{d}#1}{\mathrm{d}#2}
}

\newcommand{\ddt}[1]
{
  \diff{#1}{t}
}
